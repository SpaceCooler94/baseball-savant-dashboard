# Single requirements file used by BOTH Render (the web app) and the GitHub Action.
#
# Render needs Flask + gunicorn; the Action needs pybaseball + its data stack. Both
# environments are pinned to Python 3.11 (Render via .python-version, the Action via
# setup-python) so these versions resolve to prebuilt wheels rather than compiling
# from source.
#
# Only pandas and numpy are version-pinned, and ONLY because pybaseball 2.2.7 declares
# "pandas>=1.0.3" with no upper bound -- unpinned, pip pulls pandas 3.0 / numpy 2.5,
# which pybaseball 2.2.x breaks on. Everything else is left unpinned so pip picks
# whatever current version has a wheel for the runtime (avoids the source-compile
# failures that pinned pyarrow/lxml hit on newer Python).

Flask==3.0.3
gunicorn==22.0.0
requests==2.32.3
pybaseball==2.2.7
pandas==2.2.3
numpy==1.26.4
